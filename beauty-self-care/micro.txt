  (function () {
    const track = document.querySelector('[data-carousel-track]');
    if (!track) return;
    const slides = Array.from(track.querySelectorAll('[data-slide]'));
    const total = slides.length;
    // Start with the Derminator 2 (index 2) in the center to match the reference design.
    let active = 2;

    function render() {
      slides.forEach((slide, i) => {
        // Shortest signed distance from active on a circular track
        let diff = i - active;
        if (diff > total / 2) diff -= total;
        if (diff < -total / 2) diff += total;
        if (Math.abs(diff) <= 2) {
          slide.setAttribute('data-pos', String(diff));
        } else {
          slide.setAttribute('data-pos', 'hidden');
        }
      });
    }

    function next() { active = (active + 1) % total; render(); }
    function prev() { active = (active - 1 + total) % total; render(); }

    document.querySelector('[data-carousel-next]').addEventListener('click', next);
    document.querySelector('[data-carousel-prev]').addEventListener('click', prev);

    // Keyboard support when the carousel region is focused/hovered
    document.addEventListener('keydown', (e) => {
      if (e.key === 'ArrowRight') next();
      else if (e.key === 'ArrowLeft') prev();
    });

    // Basic touch swipe
    let startX = null;
    track.addEventListener('touchstart', (e) => { startX = e.touches[0].clientX; }, { passive: true });
    track.addEventListener('touchend', (e) => {
      if (startX === null) return;
      const dx = e.changedTouches[0].clientX - startX;
      if (Math.abs(dx) > 40) { dx < 0 ? next() : prev(); }
      startX = null;
    });

    render();
  })();
</script>

      <p>Always monitor timelines.</p>
      <h3>1–2 Weeks Before</h3>
      <ul class="clean-list">
        <li>
          <span>
            <strong>Blood thinners</strong>: avoid NSAIDs like ibuprofen
            (Advil/Motrin) and aspirin — consult your prescribing doctor if
            you're on blood thinners for a medical condition.
          </span>
        </li>        
        <li>
          <span>
            <strong>Supplements</strong>: stop fish oil, high-dose Vitamin E, 
            and herbal supplements (St. John's Wort, Ginkgo Biloba).
          </span>
        </li>
      </ul>
      <h3>3–7 Days Before</h3>
      <ul class="clean-list">
        <li><span><strong>Retinoids</strong>: stop all prescription and OTC retinol or retinoid creams (Tretinoin, Adapalene).</span></li>
        <li><span><strong>Exfoliants</strong>: halt AHAs (glycolic, lactic), BHAs (salicylic acid), facial scrubs, and at-home peel pads.</span></li>
        <li><span><strong>Acne treatments</strong>: pause benzoyl peroxide and spot treatments.</span></li>
        <li><span><strong>Other treatments</strong>: avoid waxing, laser hair removal, or chemical peels in the target area.</span></li>
      </ul>

      <h3>24–48 Hours Before</h3>
      <ul class="clean-list">
        <li>
          <span>
            <strong>Sun exposure</strong>: avoid direct, unprotected sun and tanning beds.
          </span>
        </li>
        <li>
          <span>
            <strong>Alcohol &amp; caffeine</strong>: limit alcohol (impedes healing, causes swelling); 
            skip caffeine on the day of your appointment.
          </span>
        </li>
      </ul>
      <h3>The Day Of</h3>
      <ul class="clean-list">
        <li><span><strong>Cleanse only</strong>: Start with a freshly washed face (no makeup, lotion, oils, or sunscreen).</span></li>
      </ul>

<!-- 08 · STEP-BY-STEP PROTOCOL -->
<article class="protocol" data-protocol>
  <div class="protocol__header" data-toggle>
    <div class="protocol__num">08</div>
    <div class="protocol__title-wrap">
      <h3 class="protocol__title">Step-by-Step <em>Protocol</em></h3>
      <div class="protocol__meta">Prep · treatment · recovery</div>
    </div>
    <div class="protocol__cadence">~45 minutes</div>
    <button class="protocol__toggle" aria-label="Expand protocol" aria-expanded="false">
      <svg viewBox="0 0 14 14" fill="none" stroke="currentColor" stroke-width="1">
        <path d="M7 1v12M1 7h12"/>
      </svg>
    </button>
  </div>
  <div class="protocol__body protocol__body--learn">
    <div class="protocol__body-spacer"></div>
    <div class="protocol__learn-card">
      <div class="subhead">01 · Preparation</div>
      <ol class="number-list">
        <li><span>Thoroughly cleanse face.</span></li>
        <li><span>Apply NeuroMed BLT numbing cream in an even layer.</span></li>
        <li><span>Wait 10–15 minutes for it to take effect.</span></li>
        <li><span>Remove numbing cream completely with a soft sponge.</span></li>
        <li><span>Cleanse face again with chlorhexidine.</span></li>
        <li><span>Swab the entire treatment area with an alcohol pad.</span></li>
      </ol>

      <div class="subhead">02 · Treatment</div>
      <ol class="number-list">
        <li><span>Apply hyaluronic acid serum with the dedicated silicone brush.</span></li>
        <li><span>Work in sections using a grid pattern, <em>vertical</em>, then <em>horizontal</em>, then <em>diagonal</em> passes.</span></li>
        <li><span>Keep skin taut with the non-pen hand for better, more even penetration.</span></li>
        <li><span>Use the shortest strokes possible and work quickly while numbing is active.</span></li>
        <li><span>Watch for even pinkness (erythema) as the endpoint, stop there, do not chase more.</span></li>
        <li><span>Reapply hyaluronic acid as needed for slip throughout the session.</span></li>
      </ol>

      <div class="subhead">03 · Immediate Post-Treatment</div>
      <ul class="clean-list">
        <li><span>Leave hyaluronic acid on the skin.</span></li>
        <li><span> Press in 3-5 of Rejuran Healer Turnover Ampoule over the HA while skin is still tacky. Ideal timing for PDRN support and enhanced delivery through fresh channels.
            60+ min: Seal with a ceramide moisturizer.
            Nothing else that night: no retinoids, acids, vitamin C, or actives.</span></li>

      </ul>

      <div class="subhead">04 · Next Few Days Post-Treatment</div>
      <ul class="clean-list">
        <li><span>Resume regular skincare the next day, kept bland.</span></li>
        <li><span>Wait <strong>3 days</strong> before reintroducing retinol or tretinoin.</span></li>
        <li><span>Add kiwi seed oil as the last step in the routine.</span></li>
        <li><span>Expect 3–5 days of dry, tight skin while the barrier repairs.</span></li>
      </ul>
    </div>
  </div>
</article>
